def opsbc_ranking(pareto_solutions):
    """OPSBC法によるPareto最適解の順序付け"""
    n = len(pareto_solutions)
    borda_scores = np.zeros(n)
    
    for i in range(n):
        for j in range(n):
            if dominates(pareto_solutions[i], pareto_solutions[j]):

    return np.argsort(borda_scores)[::-1]
